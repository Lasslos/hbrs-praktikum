124 gain = (kalib0!=kalib1)?100.0f/(float)(kalib1-kalib0):1.0f;
125 offset = kalib0;

127 switch( mode )

142 break;
143
144 case NXT_ULTRASCHALL:
145 maxNumOfCh = 1;
146 orb.configSensor( port, ORB::I2C, 0, 0 );
147 gain = 10.0;
148 offset = 0.0;
149 break;
150
151 case ANALOG_OFF:
152 maxNumOfCh = 1;
153 orb.configSensor( port, ORB::ANALOG, 0, 0 );
154 break;
155
156 case ANALOG_ON:
157 maxNumOfCh = 1;
158 orb.configSensor( port, ORB::ANALOG, 0, 6>>8 );
159 break;
160
161 case MB_ULTRASCHALL:
162 maxNumOfCh = 1;
163 // 30 : 420
164 // 200 : 830
165// orb.configSensor( port, ORB::TOF, 0, 0 );
166// gain = (200.0-30.0)/(830.0-420.0);
167// offset = 420-30/gain;
168 orb.configSensor( port, ORB::TOF, 0, 0 );
169 gain = 0.170; // v=2x/t => x = (0.5*0.340mm/µs)*t
170 offset = 30; // = 5mm/0.17
171 break;
172
173 case TCS3472_COLOR:
174 maxNumOfCh = 1;
