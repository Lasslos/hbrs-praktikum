145 virtual void fahre( int li, int re )
146 {
147 mLi.setSpeed( 0 );
148 mRe.setSpeed( 0 );
149
150 orb.wait( 200 );
151
152 int targetLi = mLi.getPos() + li;
153 int targetRe = mRe.getPos() + re;
154
155 // Passe Geschwindigkeit der beiden Raeder so an,
156 // dass diese gleichzeitig ihr Ziel erreichen
157 float dges = (fabs((float)li) + fabs((float)re));
158
159 mLi.setPosition( vmax * (float)li/dges, targetLi );
160 mRe.setPosition( vmax * (float)re/dges, targetRe );
161
162 orb.wait( 200 );
163
164 while( fabs( (float)mLi.getPos() - targetLi ) > 1
165 || fabs( (float)mRe.getPos() - targetRe ) > 1 )
166 {
167 }
168
169 orb.wait( 200 );
170 }
